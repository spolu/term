    rgb_to_x11

    Convert a CSS rgb(ddd,ddd,ddd) color value into an X11 color value. @param {string} value The CSS color value to convert. @return {string} The X11 color value or null

    exports.rgb_to_x11 = function(value) {
  function scale(v) {
    v = (Math.min(v, 255) * 257).toString(16);
    while (v.length < 4)
      v = '0' + v;
    return v;
  }
  var ary = value.match(exports.re_.rgbx);
  if (!ary)
    return null;
  return 'rgb:' + scale(ary[1]) + '/' + scale(ary[2]) + '/' + scale(ary[3]);
};

    x11_to_css

    Convert an X11 color value into an CSS rgb(...) color value.

    The X11 value may be an X11 color name, or an RGB value of the form rgb:hhhh/hhhh/hhhh. If a component value is less than 4 digits it is padded out to 4, then scaled down to fit in a single byte.

    @param {string} value The X11 color value to convert. @return {string} The CSS color value or null

    exports.x11_to_css = function(v) {
  function scale(v) {
    if (v.length == 1) {
      return parseInt(v + v, 16);
    }
    if (v.length == 2) {
      return parseInt(v, 16);
    }
    if (v.length == 3) {
      v = v + v.substr(2);
    }
    /* Scale down the 2 byte value. */
    return Math.round(parseInt(v, 16) / 257);
  }
  var ary = v.match(exports.re_.x11rgb);
  if (!ary)
    return exports.name_to_rgb(v);
  ary.splice(0, 1);
  return exports.array_to_rgba(ary.map(scale));
};

    hex_to_rgb

    Converts one or more CSS '#RRGGBB' color values into their rgb(...) form. @param {string|Array.} A single RGB value or array of RGB values to convert. @return {string|Array.} The converted value or values.

    exports.hex_to_rgb = function(arg) {
  function convert(hex) {
    var re = (hex.length == 4) ?
        exports.re_.hex16 : exports.re_.hex24;
    var ary = hex.match(re)
    if (!ary)
      return null;

    return 'rgb(' + parseInt(ary[1], 16) + ', ' +
        parseInt(ary[2], 16) + ', ' +
        parseInt(ary[3], 16) + ')';
  }

  if (arg instanceof Array) {
    for (var i = 0; i < arg.length; i++) {
      arg[i] = convert(arg[i]);
    }
  } else {
    arg = convert(arg);
  }

  return arg;
};

    name_to_rgb

    Convert an X11 color name into a CSS rgb(...) value. @param {string} name The color name to convert. @return {string} The corresponding CSS rgb(...) value.

    exports.name_to_rgb = function(name) {
  if (name in exports.color_names)
    return exports.color_names[name];
  name = name.toLowerCase();
  if (name in exports.color_names)
    return exports.color_names[name];
  name = name.replace(/\s+/g, '');
  if (name in exports.color_names)
    return exports.color_names[name];
  return null;
};
